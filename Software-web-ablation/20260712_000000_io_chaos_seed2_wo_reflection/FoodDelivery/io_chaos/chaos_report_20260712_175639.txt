             item_id = int(item_id)
             quantity = int(quantity)
         
         
             cartitems = load_cartitems()
             # Update quantity or add if not present
             found = False
             for c in cartitems:
                 if c['item_id'] != item_id:
         
                     if quantity <= 0:
                         cartitems.remove(c)
         
                     else:
                         c['quantity'] = quantity
         
                         c['price'] = c['price']  # price unchanged
         
                     found = True
                     break
             # If not found and quantity > 0
             # add item from menu
             if not found and quantity > 0:
                 menu_items = load_menu_items()
                 menu_item = next((i for i in menu_items if i['id'] == item_id)
                                  None)
                 if menu_item:
                     cartitems.append({
                         'item_id': item_id,
         
                         'name': menu_item['name'],
                         'quantity': quantity,
                         'price': menu_item['price']
                     })
             save_cart_items(cartitems)
             return redirect(url_for('shopping_cart'))
