         @app.route('/cart', methods=['POST'])
         def cart():
             cart = read_cart()
             books = read_books()
         
             cart_items = []
             total_amount = 0
         
             # Map book_id to title and price
             book_map = {}
             for b in books:
                 book_map[b['book_id']] = (b['title'], b['price'])
         
             for item in cart:
                 try:
                     item_id = int(item['cart_id'])
                     book_id = item['book_id']
                     quantity = int(item['quantity'])
                     if book_id in book_map:
                         title, price_str = book_map[book_id]
                         try:
                             price = float(price_str)
                         except Exception:
                             price = 0
                         subtotal = price * quantity
                         total_amount += subtotal
         
                         cart_items.append({
                             'item_id': item_id,
                             'book_title': title,
                             'quantity': str(quantity),
                             'price': price,
                             'subtotal': subtotal
                         })
                 except Exception:
                     continue
         
             total_amount_int = int(total_amount) if total_amount == int(total_amount) else total_amount
         
             return render_template('cart.html', cart_items=cart_items, total_amount=total_amount_int)

             return redirect(url_for('cart'))

         @app.route('/checkout', methods=['GET', 'POST'])
         def checkout():
             if request.method == 'GET':
                 cart = read_cart()
                 books = read_books()
         
                 cart_items = []
                 total_amount = 0
         
                 book_map = {}
                 for b in books:
                     book_map[b['book_id']] = (b['title'], b['price'])
         
                 for item in cart:
                     try:
                         item_id = int(item['cart_id'])
                         book_id = item['book_id']
                         quantity = int(item['quantity'])
                         if book_id in book_map:
                             title, price_str = book_map[book_id]
                             try:
                                 price = float(price_str)
                             except Exception:
                                 price = 0
                             subtotal = price * quantity
                             total_amount += subtotal
         
                             cart_items.append({
                                 'item_id': item_id,
                                 'book_title': title,
                                 'quantity': str(quantity),
                                 'price': price,
                                 'subtotal': subtotal
                             })
                     except Exception:
                         continue
         
                 total_amount_int = int(total_amount) if total_amount == int(total_amount) else total_amount
         
                 return render_template('checkout.html', cart_items=cart_items, total_amount=total_amount_int)
             else:
                 # POST: process checkout form (shipping and payment info)
                 customer_name = request.form.get('customername')
                 shipping_addr = request.form.get('ship_address')
                 pay_method = request.form.get('pay_method')
         
                 if not customer_name or not shipping_addr or not pay_method:
                     # Missing info, redisplay form
                     return render_template('checkout.html', error='Missing required information')
         
                 # Normally would do more validation on payment method etc.
                 # Save order
                 cart = read_cart()
                 if not cart:
                     return redirect(url_for('cart'))
         
                 books = read_books()
                 book_map = {b['book_id']: b for b in books}
         
                 orders = read_orders()
                 order_items = read_order_items()
         
                 # New order ID
                 max_order_id = 0
                 for o in orders:
                     try:
                         oid = int(o['order_id'])
                         if oid > max_order_id:
                             max_order_id = oid
                     except Exception:
                         continue
                 new_order_id = max_order_id + 1
         
                 # total amount calculation
                 total_amount = 0
                 for item in cart:
                     book_id = item['book_id']
                     quantity = int(item['quantity'])
                     price_str = book_map.get(book_id, {}).get('price', '0')
                     try:
                         price = float(price_str)
                     except Exception:
                         price = 0
                     total_amount += price * quantity
         
                 total_amount_str = str(total_amount)
                 today_str = datetime.utcnow().strftime('%Y/%m/%d')
         
                 # Add order
                 new_order = {
                     'order_id': str(new_order_id),
                     'customer_name': customer_name,
                     'order_date': today_str,
                     'total_amount': total_amount_str,
                     'status': 'Pending',
                     'shipping_address': shipping_addr
                 }
         
                 orders.append(new_order)
         
                 # Add order_items
                 max_order_item_id = 0
                 for oi in order_items:
                     try:
                         oid = int(oi['order_item_id'])
                         if oid > max_order_item_id:
                             max_order_item_id = oid
                     except Exception:
                         continue
         
                 for item in cart:
                     max_order_item_id += 1
                     book_id = item['book_id']
                     quantity = item['quantity']
                     price_str = book_map.get(book_id, {}).get('price', '0')
                     try:
                         price = float(price_str)
                     except Exception:
                         price = 0
                     order_items.append({
                         'order_item_id': str(max_order_item_id),
                         'order_id': str(new_order_id),
                         'book_id': book_id,
                         'quantity': quantity,
                         'price': price
                     })
         
                 write_orders(orders)
                 write_order_items(order_items)
         
                 # Clear cart
                 write_cart([])
         
                 return redirect(url_for('orders'))

         @app.route('/checkout', methods=['GET', 'POST'])
         def checkout():
             if request.method == 'GET':
                 cart = read_cart()
                 books = read_books()
         
                 cart_items = []
                 total_amount = 0
         
                 book_map = {}
                 for b in books:

                     try:
                         item_id = int(item['cart_id'])

                         if book_id in book_map:
                             title, price_str = book_map[book_id]
                             try:
                                 price = float(price_str)
                             except Exception:
                                 price = 0
                             subtotal = price * quantity
                             total_amount += subtotal

                                 'item_id': item_id,
                                 'book_title': title,
                                 'quantity': str(quantity),
                                 'price': price,
                                 'subtotal': subtotal
                             })
                     except Exception:
                         continue
         
                 total_amount_int = int(total_amount) if total_amount == int(total_amount) else total_amount

             else:
                 # POST: process checkout form (shipping and payment info)
                 customer_name = request.form.get('customername')
                 shipping_addr = request.form.get('ship_address')
                 pay_method = request.form.get('pay_method')
         
                 if not customer_name or not shipping_addr or not pay_method:
                     # Missing info, redisplay form
                     return render_template('checkout.html', error='Missing required information')
         
                 # Normally would do more validation on payment method etc.
                 # Save order
                 cart = read_cart()
                 if not cart:
                     return redirect(url_for('cart'))
         
                 books = read_books()
                 book_map = {b['book_id']: b for b in books}
         
                 orders = read_orders()
                 order_items = read_order_items()
         
                 # New order ID
                 max_order_id = 0
                 for o in orders:
                     try:
                         oid = int(o['order_id'])
                         if oid > max_order_id:
                             max_order_id = oid
                     except Exception:
                         continue
                 new_order_id = max_order_id + 1
         
                 # total amount calculation
                 total_amount = 0
                 for item in cart:
                     book_id = item['book_id']
                     quantity = int(item['quantity'])
                     price_str = book_map.get(book_id, {}).get('price', '0')
                     try:
                         price = float(price_str)
                     except Exception:
                         price = 0
                     total_amount += price * quantity
         
                 total_amount_str = str(total_amount)
                 today_str = datetime.utcnow().strftime('%Y/%m/%d')
         
                 # Add order
                 new_order = {
                     'order_id': str(new_order_id),
                     'customer_name': customer_name,
                     'order_date': today_str,
                     'total_amount': total_amount_str,
                     'status': 'Pending',
                     'shipping_address': shipping_addr
                 }
         
                 orders.append(new_order)
         
                 # Add order_items
                 max_order_item_id = 0
                 for oi in order_items:
                     try:
                         oid = int(oi['order_item_id'])
                         if oid > max_order_item_id:
                             max_order_item_id = oid
                     except Exception:
                         continue

                     max_order_item_id += 1

                         price = float(price_str)

                         'book_id': book_id,
                         'quantity': quantity,
                         'price': price
                     })
         
                 write_orders(orders)
                 write_order_items(order_items)
         
                 # Clear cart
                 write_cart([])
