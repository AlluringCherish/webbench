                             continue
                         parts = line.split('|')
                         if len(parts) != 6:
                             continue
                         resume = {
                             'resume_id': int(parts[0]),
                             'applicant_name': parts[1],
                             'applicant_email': parts[2],
                             'filename': parts[3],
                             'upload_date': parts[4],
                             'summary': parts[5]
                         }
                         resumes.append(resume)
             except FileNotFoundError:
                 pass
             return resumes
