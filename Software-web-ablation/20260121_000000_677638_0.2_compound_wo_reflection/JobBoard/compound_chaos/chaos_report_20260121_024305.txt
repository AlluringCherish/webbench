         # Jobs Route
         @app.route('/jobs', methods=['GET', 'POST'])
         def jobs():
             jobs_list = read_jobs()
             companies = read_companies()
             # For POST filter/search
             search_query = ''
             filter_type = ''
             filter_location = ''
         
             if request.method == 'POST':
                 search_query = request.form.get('search_query', '').strip().lower()
                 filter_type = request.form.get('filter_type', '')
                 filter_location = request.form.get('filter_location', '')
         
                 filtered = []
                 for job in jobs_list:
                     # Filter type (category)
                     if filter_type and job['category'] != filter_type:
                         continue
                     # Filter location
                     if filter_location and job['location'] != filter_location:
                         continue
                     # Search query in title or description case insensitive
                     if search_query:
                         if search_query not in job['title'].lower() and search_query not in job['description'].lower():
                             continue
                     filtered.append(job)
                 jobs_list = filtered
         
             # Prepare job_cards context list
             job_cards = []
             for job in jobs_list:
                 # get company name
                 company = get_company_by_id(job['company_id'])
                 company_name = company['name'] if company else 'Unknown'
                 salary_range = f"${job['salary_min']:,} - ${job['salary_max']:,}"
                 job_cards.append({
                     'job_id': job['job_id'],
                     'title': job['title'],
                     'company': company_name,
                     'salary_range': salary_range
                 })
         
             return render_template('jobs.html',
                                    search_input_id='search-input',
                                    filter_type_dropdown_id='filter-type-dropdown',
                                    filter_location_dropdown_id='filter-location-dropdown',
                                    job_cards=job_cards)

         # Submit Application Route
         @app.route('/apply/<int:job_id>', methods=['GET', 'POST'])
         def submit_application(job_id):
             job = get_job_by_id(job_id)
             if not job:
                 return redirect(url_for('jobs'))
         
             error = None
         
             if request.method == 'POST':
                 applicant_name = request.form.get('applicant_name', '').strip()
                 applicant_email = request.form.get('applicant_email', '').strip()
                 # For simplicity store resume content from cover_letter field as resume content,
                 # as file upload handling not specified and no files directory to save
                 # So form field 'cover_letter' and 'resume-upload' handled for validation only
                 cover_letter = request.form.get('cover_letter', '').strip()
         
                 # Validate minimal inputs
                 if not applicant_name or not applicant_email:
                     error = 'Name and Email are required.'
         
                 if error is None:
                     # Add a new resume entry
                     resumes = read_resumes()
                     new_resume_id = (max([r['resume_id'] for r in resumes]) + 1) if resumes else 1
                     new_resume = {
                         'resume_id': new_resume_id,
                         'applicant_name': applicant_name,
                         'content': cover_letter
                     }
                     resumes.append(new_resume)
                     write_resumes(resumes)
         
                     # Add a new application
                     applications = read_applications()
                     new_application_id = (max([a['application_id'] for a in applications]) + 1) if applications else 1
                     today = datetime.now().strftime('%Y-%m-%d')
                     new_application = {
                         'application_id': new_application_id,
                         'job_id': job_id,
                         'applicant_name': applicant_name,
                         'applicant_email': applicant_email,
                         'status': 'Applied',
                         'applied_date': today,
                         'resume_id': new_resume_id
                     }
                     applications.append(new_application)
                     write_applications(applications)
         
                     # On success, clear input fields (achieved by redirecting back to form GET to get empty form)
                     return redirect(url_for('submit_application', job_id=job_id))
         
             return render_template('application_form.html',
                                    form_container_id='application-form-container',
                                    applicant_name_id='applicant-name',
                                    applicant_email_id='applicant-email',
                                    resume_upload_id='resume-upload',
                                    cover_letter_id='cover-letter',
                                    submit_button_id='submit-application-button',
                                    job=job,
                                    error=error)
