from flask import Flask, render_template, redirect, url_for, request

data_dir = 'data'

### Utility functions for data handling ###

def load_users():
    users = []
    try:
        with open(os.path.join(data_dir, 'users.txt'), 'r') as f:
            for line in f:
                parts = line.strip().split('|')
                if len(parts) == 3:
                    user = {'username': parts[0], 'email': parts[1], 'password': parts[2]}
                    users.append(user)
    except FileNotFoundError:
        pass
    return users

def save_users(users):
    with open(os.path.join(data_dir, 'users.txt'), 'w') as f:
        for user in users:
            line = '|'.join([user['username'], user['email'], user['password']])
            f.write(line + '\n')

def load_auctions():
    auctions = []
    try:
        with open(os.path.join(data_dir, 'auctions.txt'), 'r') as f:
            for line in f:
                parts = line.strip().split('|')
                if len(parts) == 5:
                    auction = {
                        'auction_id': parts[0],
                        'item_name': parts[1],
                        'description': parts[2],
                        'start_price': float(parts[3]),
                        'owner_username': parts[4]
                    }
                    auctions.append(auction)
    except FileNotFoundError:
        pass
    return auctions

def save_auctions(auctions):
    with open(os.path.join(data_dir, 'auctions.txt'), 'w') as f:
        for auction in auctions:
            line = '|'.join([
                auction['auction_id'],
                auction['item_name'],
                auction['description'],
                str(auction['start_price']),
                auction['owner_username']
            ])
            f.write(line + '\n')

def load_bids():
    bids = []
    try:
        with open(os.path.join(data_dir, 'bids.txt'), 'r') as f:
            for line in f:
                parts = line.strip().split('|')
                if len(parts) == 4:
                    bid = {
                        'bid_id': parts[0],
                        'auction_id': parts[1],
                        'bidder_username': parts[2],

                    }

        pass

@app.route('/login', methods=['GET', 'POST'])
def login():
    error = None
    if request.method == 'POST':
        username = request.form.get('username', '')
        password = request.form.get('password', '')
        users = load_users()
        user = next((u for u in users if u['username'] == username and u['password'] == password), None)
        if user:
            # For demonstration, we don't implement sessions or login state
            return redirect(url_for('dashboard'))
        else:
            error = 'Invalid credentials'
    return render_template('login.html', error=error)

@app.route('/register', methods=['GET', 'POST'])
def register():
    error = None
    if request.method == 'POST':
        username = request.form.get('username', '')
        email = request.form.get('email', '')
        password = request.form.get('password', '')
        users = load_users()
        if any(u['username'] == username for u in users):
            error = 'Username already exists'

@app.route('/auction/<auction_id>')
def auction_detail(auction_id):
    auctions = load_auctions()
    auction = next((a for a in auctions if a['auction_id'] == auction_id), None)
    if not auction:
        return "Auction not found", 404
    bids = load_bids()
    auction_bids = [b for b in bids if b['auction_id'] == auction_id]
    return render_template('auction_detail.html', auction=auction, bids=auction_bids)

@app.route('/auction/<auction_id>/bid', methods=['POST'])
def bid(auction_id):
    bidder_username = request.form.get('username', '')
    bid_amount_str = request.form.get('bid_amount', '')
    try:
        bid_amount = float(bid_amount_str)
    except ValueError:
        bid_amount = None
    auctions = load_auctions()
    auction = next((a for a in auctions if a['auction_id'] == auction_id), None)
    if not auction or not bidder_username or bid_amount is None or bid_amount <= 0:
        return redirect(url_for('auction_detail', auction_id=auction_id))
    bids = load_bids()
    # Generate new bid_id
    if bids:
        max_id = max(int(b['bid_id']) for b in bids)
    else:
        max_id = 0
    new_bid = {
        'bid_id': str(max_id + 1),
        'auction_id': auction_id,
        'bidder_username': bidder_username,
        'bid_amount': bid_amount
    }
    bids.append(new_bid)
    save_bids(bids)
    return redirect(url_for('auction_detail', auction_id=auction_id))
