         @app.route('/dashboard')
         def dashboard_page():
             songs = load_songs()
             artists = load_artists()
         
             # featured_songs: list of dict {song_id: int, title: str, artist_name: str, duration: int}
             # We pick top 5 songs by play_count for featured
             sorted_songs = sorted(songs, key=lambda x: x.get('play_count',0), reverse=True)
             featured_songs = []
             for s in sorted_songs[:5]:
                 artist_name = get_artist_name(s['artist_id'], artists)
                 featured_songs.append({
                     'song_id': s['song_id'],
                     'title': s['title'],
                     'artist_name': artist_name,
                     'duration': s['duration']
                 })
         
             # trending_artists: list of dict {artist_id: int, name: str}
             # Pick top 5 artists by sum play_count for their songs
             artist_play_counts = {}
             for s in songs:
                 artist_play_counts[s['artist_id']] = artist_play_counts.get(s['artist_id'], 0) + s.get('play_count', 0)
             sorted_artists_play = sorted(artist_play_counts.items(), key=lambda i:i[1], reverse=True)
         
             trending_artists = []
             for artist_id,_ in sorted_artists_play[:5]:
                 artist = get_artist(artist_id, artists)
                 if artist:
                     trending_artists.append({'artist_id': artist['artist_id'], 'name': artist['name']})
         
             return render_template('dashboard.html', featured_songs=featured_songs, trending_artists=trending_artists)

                     parts = line.split('|')
                     if len(parts) != 3:
                         continue
                     try:
                         genre = {
                             'genre_id': int(parts[0]),
                             'genre_name': parts[1],
                             'description': parts[2]
                         }
                         genres.append(genre)
                     except ValueError:
                         continue
             return genres
         
         # Load Playlists
         
         def load_playlists():
             playlists = []
             filepath = os.path.join(data_folder, 'playlists.txt')
             if not os.path.exists(filepath):
                 return playlists
             with open(filepath, encoding='utf-8') as f:
                 for line in f:
                     line = line.strip()
                     if not line:
                         continue
                     parts = line.split('|')
                     if len(parts) != 5:
                         continue
                     try:
                         playlist = {
                             'playlist_id': int(parts[0]),
                             'title': parts[1],
                             'description': parts[2],
                             'creation_date': parts[3],
                             'total_songs': int(parts[4])
                         }
                         playlists.append(playlist)
                     except ValueError:
                         continue
             return playlists
         
         # Load Playlist Songs
         
         def load_playlist_songs():
             playlist_songs = []
             filepath = os.path.join(data_folder, 'playlist_songs.txt')
             if not os.path.exists(filepath):
                 return playlist_songs
             with open(filepath, encoding='utf-8') as f:
                 for line in f:
                     line = line.strip()
                     if not line:
                         continue
                     parts = line.split('|')
                     if len(parts) != 4:
                         continue
                     try:
                         pls = {
                             'playlist_song_id': int(parts[0]),
                             'playlist_id': int(parts[1]),
                             'song_id': int(parts[2]),
                             'added_date': parts[3]
                         }
                         playlist_songs.append(pls)
                     except ValueError:
                         continue
             return playlist_songs
         
         # Save Playlists
         
         def save_playlists(playlists):
             filepath = os.path.join(data_folder, 'playlists.txt')
             with open(filepath, 'w', encoding='utf-8') as f:
                 for pl in playlists:
                     line = f"{pl['playlist_id']}|{pl['title']}|{pl['description']}|{pl['creation_date']}|{pl['total_songs']}\n"
                     f.write(line)
         
         # Save Playlist Songs
         
         def save_playlist_songs(playlist_songs):
             filepath = os.path.join(data_folder, 'playlist_songs.txt')

                 for pls in playlist_songs:
                     line = f"{pls['playlist_song_id']}|{pls['playlist_id']}|{pls['song_id']}|{pls['added_date']}\n"
                     f.write(line)

         def get_next_playlist_id(playlists):
             max_id = 0

                     max_id = pl['playlist_id']
             return max_id + 1

         @app.route('/')
         def dashboard_page():
             songs = load_songs()
             artists = load_artists()
             # Top 5 songs by play count
             sorted_songs = sorted(songs, key=lambda x: x['play_count'], reverse=True)[:5]
             top_songs = []
             for s in sorted_songs:
                 artist_name = get_artist_name(s['artist_id'], artists)
                 top_songs.append({'song_id': s['song_id'], 'title': s['title'], 'artist_name': artist_name, 'play_count': s['play_count']})
         
             # Trending artists by total play counts
             artist_play_counts = {}
             for s in songs:
                 artist_play_counts[s['artist_id']] = artist_play_counts.get(s['artist_id'], 0) + s['play_count']
             artists_list = load_artists()
             trending_artists = []
             for artist in artists_list:
                 count = artist_play_counts.get(artist['artist_id'], 0)
                 trending_artists.append({'artist_id': artist['artist_id'], 'name': artist['name'], 'total_play_count': count})
             trending_artists.sort(key=lambda x: x['total_play_count'], reverse=True)
         
             return render_template('dashboard.html', top_songs=top_songs, trending_artists=trending_artists)
