def load_auctions():
    auctions = []
    path = os.path.join(DATA_DIR, 'auctions.txt')
    if not os.path.exists(path):
        return auctions
    with open(path, 'r', encoding='utf-8') as f:
        for line in f:
            parts = line.strip().split('|')
            if len(parts) < 9:
                continue
            auction = {
                'auction_id': int(parts[0]),
                'item_name': parts[1],
                'description': parts[2],
                'category': parts[3],
                'starting_bid': float(parts[4]),
                'current_bid': float(parts[5]),
                'end_time': parts[6],
                'status': parts[7],
                'image_url': parts[8]
            }
            auctions.append(auction)
    return auctions

@app.route('/auction/<int:auction_id>/place_bid', methods=['POST'])
def submit_bid(auction_id):
    auctions = load_auctions()
    bids = load_bids()

    auction = next((a for a in auctions if a['auction_id'] == auction_id), None)
    if not auction:
        return "Auction not found", 404

    bidder_name = request.form.get('bidder_name', '').strip()
    bid_amount_raw = request.form.get('bid_amount', '').strip()

    errors = {}
    if not bidder_name:
        errors['bidder_name'] = 'Bidder name is required.'

    try:
        bid_amount = float(bid_amount_raw)
    except (ValueError, TypeError):
        errors['bid_amount'] = 'Invalid bid amount.'
        bid_amount = None

    minimum_bid = auction['current_bid'] if auction['current_bid'] > auction['starting_bid'] else auction['starting_bid']

    if bid_amount is not None and bid_amount < (minimum_bid + 0.01):  # Must bid higher than current
        errors['bid_amount'] = f'Bid amount must be at least ${minimum_bid + 0.01:.2f}.'

    if errors:
        auction_info = {
            'auction_id': auction['auction_id'],
            'item_name': auction['item_name'],
            'minimum_bid': minimum_bid
        }
        return render_template('place_bid.html', auction=auction_info, bidder_name=bidder_name, bid_amount=bid_amount_raw, errors=errors)

    # Add the bid
    new_bid_id = max([b['bid_id'] for b in bids], default=0) + 1

    # Use current time as bid_timestamp in format YYYY-MM-DD HH:MM (simple way)
    from datetime import datetime
    bid_timestamp = datetime.now().strftime('%Y-%m-%d %H:%M')

    new_bid = {
        'bid_id': new_bid_id,
        'auction_id': auction_id,
        'bidder_name': bidder_name,
        'bid_amount': bid_amount,
        'bid_timestamp': bid_timestamp
    }
    bids.append(new_bid)
    save_bids(bids)

    # Update auction current_bid if bid is higher
    if bid_amount > auction['current_bid']:
        auction['current_bid'] = bid_amount
        save_auctions(auctions)

    return redirect(url_for('auction_details', auction_id=auction_id))

    bid_amount_raw = request.form.get('bid_amount', '').strip()

    errors = {}
    if not bidder_name:
        errors['bidder_name'] = 'Bidder name is required.'

    try:
        bid_amount = float(bid_amount_raw)
    except (ValueError, TypeError):
        errors['bid_amount'] = 'Invalid bid amount.'
        bid_amount = None

    minimum_bid = auction['current_bid'] if auction['current_bid'] > auction['starting_bid'] else auction['starting_bid']

    if bid_amount is not None and bid_amount < (minimum_bid + 0.01):  # Must bid higher than current
        errors['bid_amount'] = f'Bid amount must be at least ${minimum_bid + 0.01:.2f}.'

    if errors:
        auction_info = {
            'auction_id': auction['auction_id'],
            'item_name': auction['item_name'],
            'minimum_bid': minimum_bid
        }
        return render_template('place_bid.html', auction=auction_info, bidder_name=bidder_name, bid_amount=bid_amount_raw, errors=errors)

    # Add the bid
    new_bid_id = max([b['bid_id'] for b in bids], default=0) + 1

    # Use current time as bid_timestamp in format YYYY-MM-DD HH:MM (simple way)
    from datetime import datetime
    bid_timestamp = datetime.now().strftime('%Y-%m-%d %H:%M')

    new_bid = {
        'bid_id': new_bid_id,
        'auction_id': auction_id,
        'bidder_name': bidder_name,
        'bid_amount': bid_amount,
        'bid_timestamp': bid_timestamp
    }
    bids.append(new_bid)
    save_bids(bids)

    # Update auction current_bid if bid is higher
    if bid_amount > auction['current_bid']:
        auction['current_bid'] = bid_amount
        save_auctions(auctions)

    return redirect(url_for('auction_details', auction_id=auction_id))
