         @app.route('/cart', methods=['GET', 'POST'])
         def shopping_cart():
             menus = load_menus()
             cart_items = load_cart()
         
             if request.method == 'POST':
                 # Possible actions: update quantities, remove items
                 # We parse form keys for these operations
                 new_cart_items = cart_items.copy()
                 action_handled = False
         
                 # Update quantities or remove items based on form
                 for key in request.form:
                     # Remove item: remove-item-button-{item_id}
                     if key.startswith('remove-item-button-'):
                         try:
                             item_id = int(key[len('remove-item-button-'):])
                             new_cart_items = [ci for ci in new_cart_items if ci['item_id'] != item_id]
                             action_handled = True
                         except ValueError:
                             pass
                     # Update quantity: update-quantity-{item_id}
                     elif key.startswith('update-quantity-'):
                         try:
                             item_id = int(key[len('update-quantity-'):])
                             quantity_str = request.form.get(key, '').strip()
                             if quantity_str.isdigit():
                                 quantity = int(quantity_str)
                                 if quantity <= 0:
                                     # remove item if quantity zero or less
                                     new_cart_items = [ci for ci in new_cart_items if ci['item_id'] != item_id]
                                 else:
                                     # update quantity for item
                                     found = False
                                     for ci in new_cart_items:
                                         if ci['item_id'] == item_id:
                                             ci['quantity'] = quantity
                                             found = True
                                             break
                                     if not found:
                                         # ignore, item not found in cart
                                         pass
                                 action_handled = True
                         except ValueError:
                             pass
         
                 if action_handled:
                     # Save updated cart
                     # Fix cart_id and added_date: preserve cart_id, added_date of existing or assign new
                     existing_ids = {ci['cart_id'] for ci in new_cart_items}
                     next_cart_id = max(existing_ids) + 1 if existing_ids else 1
                     for ci in new_cart_items:
                         if 'cart_id' not in ci or ci['cart_id'] is None:
                             ci['cart_id'] = next_cart_id
                             ci['added_date'] = datetime.now().strftime('%Y-%m-%d')
                             next_cart_id += 1
         
                     save_cart(new_cart_items)
         
                 # After POST save, reload updated cart_items
                 cart_items = load_cart()
         
             # Prepare cart items with menu info for template
             detailed_cart_items = []
             total_amount = 0.0
             for ci in cart_items:
                 menu_item = next((m for m in menus if m['item_id'] == ci['item_id']), None)
                 if menu_item:
                     subtotal = menu_item['price'] * ci['quantity']
                     total_amount += subtotal
                     item_dict = {
                         'cart_id': ci['cart_id'],
                         'item_id': ci['item_id'],
                         'restaurant_id': ci['restaurant_id'],
                         'item_name': menu_item['item_name'],
                         'quantity': ci['quantity'],
                         'price': menu_item['price'],
                         'subtotal': subtotal
                     }
                     detailed_cart_items.append(item_dict)
         
             return render_template('cart.html', cart_items=detailed_cart_items, total_amount=total_amount)
         
         
         @app.route('/checkout', methods=['GET', 'POST'])
         def checkout():
             # Fields: customer_name, delivery_address, phone_number, payment_method
             form_fields = ['customer_name', 'delivery_address', 'phone_number', 'payment_method']
             errors = {}
             order_form = {field: '' for field in form_fields}

                     errors['payment_method'] = 'Payment method is required.'
         
                 if not errors:
                     # Load cart
                     cart_items = load_cart()

                             if not menu_item:
                                 errors['cart'] = f"Item {ci['item_id']} is no longer available."
                                 break
                             total += menu_item['price'] * ci['quantity']
         
                         if not errors:
                             orders = load_orders()

                             existing_order_ids = [o['order_id'] for o in orders]
                             new_order_id = max(existing_order_ids) + 1 if existing_order_ids else 1

                                 order_date = datetime.now().strftime('%Y-%m-%d')
         
                                 # Create new order record

                                     'customer_name': order_form['customer_name'],

                                     'status': 'Pending',
                                     'delivery_address': order_form['delivery_address'],
                                     'phone_number': order_form['phone_number']
                                 }
         
                                 # Append to orders file
                                 try:
                                     with open(ORDERS_FILE, 'a', encoding='utf-8') as f:
                                         line = f"{new_order['order_id']}|{new_order['customer_name']}|{new_order['restaurant_id']}|{new_order['order_date']}|{new_order['total_amount']:.2f}|{new_order['status']}|{new_order['delivery_address']}|{new_order['phone_number']}
         "
                                         f.write(line)
                                 except IOError:
                                     errors['file'] = 'Could not save order. Please try again later.'
         
                                 if 'file' not in errors:
                                     # Save order items
                                     order_items = load_order_items()
                                     existing_order_item_ids = [oi['order_item_id'] for oi in order_items]
                                     next_order_item_id = max(existing_order_item_ids) + 1 if existing_order_item_ids else 1
         
                                     try:
                                         with open(ORDER_ITEMS_FILE, 'a', encoding='utf-8') as f:

                                                 line = f"{next_order_item_id}|{new_order_id}|{menu_item['item_id']}|{ci['quantity']}|{menu_item['price']:.2f}
         "
                                                 f.write(line)
                                                 next_order_item_id += 1

                                         errors['file'] = 'Could not save order items. Please try again later.'

                                                 pass  # clear file

         ## 4. Data Storage
         
         The 'FoodDelivery' application will store data locally in text files organized in the directory 'data'. The following data formats and examples are defined:
         
         ### 1. Restaurants Data
         - **File Name**: `restaurants.txt`
         - **Data Format**:
           ```
           restaurant_id|name|cuisine|address|phone|rating|delivery_time|min_order
           ```
         - **Example Data**:
           ```
           1|Dragon House|Chinese|123 Main St|555-0001|4.5|30|15.00
           2|La Bella Italia|Italian|456 Oak Ave|555-0002|4.8|25|20.00
           3|Taj Mahal|Indian|789 Elm St|555-0003|4.6|35|18.00
           ```
         
         ### 2. Menus Data
         - **File Name**: `menus.txt`
         - **Data Format**:
           ```
           item_id|restaurant_id|item_name|category|description|price|availability
           ```
         - **Example Data**:
           ```
           1|1|Fried Rice|Main Course|Steamed rice with vegetables and egg|12.99|1
           2|1|Spring Rolls|Appetizer|Crispy rolls with pork filling|8.99|1
           3|2|Spaghetti Carbonara|Pasta|Classic Italian pasta with cream sauce|14.99|1
           ```
         
         ### 3. Cart Data
         - **File Name**: `cart.txt`
         - **Data Format**:
           ```
           cart_id|item_id|restaurant_id|quantity|added_date
           ```
         - **Example Data**:
           ```
           1|1|1|2|2025-01-15
           2|3|2|1|2025-01-16
           ```
         
         ### 4. Orders Data
         - **File Name**: `orders.txt`
         - **Data Format**:
           ```
           order_id|customer_name|restaurant_id|order_date|total_amount|status|delivery_address|phone_number
           ```
         - **Example Data**:
           ```
           1|John Doe|1|2025-01-10|21.98|Delivered|123 Main St, NYC|555-1234
           2|Jane Smith|2|2025-01-14|14.99|On the Way|456 Oak Ave, LA|555-5678
           ```
         
         ### 5. Order Items Data
         - **File Name**: `order_items.txt`
         - **Data Format**:
           ```
           order_item_id|order_id|item_id|quantity|price
           ```
         - **Example Data**:
           ```
           1|1|1|2|12.99
           2|1|2|1|8.99
           3|2|3|1|14.99
           ```
         
         ### 6. Deliveries Data
         - **File Name**: `deliveries.txt`
         - **Data Format**:
           ```
           delivery_id|order_id|driver_name|driver_phone|vehicle_info|status|estimated_time
           ```
         - **Example Data**:
           ```
           1|1|Mike Johnson|555-9001|Bike|Delivered|2025-01-10 18:45
           2|2|Sarah Williams|555-9002|Car|On the Way|2025-01-14 19:30
           ```
         
         ### 7. Reviews Data
         - **File Name**: `reviews.txt`
         - **Data Format**:
           ```
           review_id|restaurant_id|customer_name|rating|review_text|review_date
           ```
         - **Example Data**:
           ```
           1|1|Alice Johnson|5|Excellent food and fast delivery!|2025-01-12
           2|2|Bob Williams|4|Great pasta, slightly delayed delivery.|2025-01-13
           3|3|Charlie Brown|5|Best Indian food in town!|2025-01-15
           ```
